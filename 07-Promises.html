function pegarTemperatura() {
            return new Promise(function (resolve, reject) {
                console.log('Consultando temperatura...');

                setTimeout(function() {
                    resolve('40 na sombra');
                }, 2000);
            });
        }

        // 3 resultado possiveis:
        //      pendente, ou seja, estou esperando
        //      finalizado, deu tudo certo
        //      rejeitado, aconteceu algum problema

        // Usando a promise
        let temp = pegarTemperatura();
        // console.log(temp);
        
        // Quando deu tudo certo: callback. mecanismo de qdo der tudo certo
        temp.then(function(resultado) {
           console.log('Temperatura: ' + resultado); 
        });
        
        // Quando deu tudo errado: outro callback
        temp.catch(function(error){
            console.log("Eita, deu erro!");
        });
